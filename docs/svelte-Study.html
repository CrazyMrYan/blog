	function add(input) {
		const todo = {
			id: uid++,
			done: false,
			description: input.value
		};

		todos = [todo, ...todos];
		input.value = '';
	}
	// åˆ é™¤æ–¹æ³•
	function remove(todo) {
		todos = todos.filter(t => t !== todo);
	}
	// é€‰ä¸­æ–¹æ³•
	function mark(todo, done) {
		todo.done = done;
		remove(todo);
		todos = todos.concat(todo);
	}
</script>
